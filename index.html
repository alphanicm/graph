        // 1. 수학적 계산 헬퍼 함수

        const factCache = [1, 1];

        function factorial(num) {

            if (factCache[num]) return factCache[num];

            let result = factCache[factCache.length - 1];

            for (let i = factCache.length; i <= num; i++) {

                result *= i;

                factCache[i] = result;

            }

            return result;

        }

        function calcExpTaylor(x, n) { let sum = 0; for (let k = 0; k <= n; k++) sum += Math.pow(x, k) / factorial(k); return sum; }

        function calcSinTaylor(x, n) { let sum = 0; for (let k = 0; k <= n; k++) if (k % 2 !== 0) sum += ((Math.floor((k - 1) / 2) % 2 === 0) ? 1 : -1) * Math.pow(x, k) / factorial(k); return sum; }

        function calcCosTaylor(x, n) { let sum = 0; for (let k = 0; k <= n; k++) if (k % 2 === 0) sum += ((Math.floor(k / 2) % 2 === 0) ? 1 : -1) * Math.pow(x, k) / factorial(k); return sum; }

        function getOriginalY(func, xVals) {

            if (func === 'sin') return xVals.map(x => Math.sin(x));

            if (func === 'cos') return xVals.map(x => Math.cos(x));

            if (func === 'exp') return xVals.map(x => Math.exp(x));

        }

        function getSmoothTaylorY(func, continuousN, xVals) {

            const n1 = Math.floor(continuousN);

            const n2 = Math.min(20, n1 + 1);

            const r = continuousN - n1;

            return xVals.map(x => {

                let y1, y2;

                if (func === 'sin') { y1 = calcSinTaylor(x, n1); y2 = calcSinTaylor(x, n2); }

                else if (func === 'cos') { y1 = calcCosTaylor(x, n1); y2 = calcCosTaylor(x, n2); }

                else { y1 = calcExpTaylor(x, n1); y2 = calcExpTaylor(x, n2); }

                return (1 - r) * y1 + r * y2;

            });

        }

        // 2. 수식(LaTeX) 문자열 생성 함수

        function generatePolynomialString(func, n) {

            let terms = [];

            if (func === 'exp') {

                for (let k = 0; k <= n; k++) terms.push(k === 0 ? '1' : (k === 1 ? '+ x' : `+ \\frac{x^{${k}}}{${k}!}`));

            } else if (func === 'sin') {

                for (let k = 0; k <= n; k++) if (k % 2 !== 0) terms.push(`${(Math.floor((k - 1) / 2) % 2 === 0) ? '+' : '-'} ${k === 1 ? 'x' : `\\frac{x^{${k}}}{${k}!}`}`);

                if (terms.length > 0 && terms[0].startsWith('+ ')) terms[0] = terms[0].substring(2);

            } else if (func === 'cos') {

                for (let k = 0; k <= n; k++) if (k % 2 === 0) terms.push(`${(Math.floor(k / 2) % 2 === 0) ? '+' : '-'} ${k === 0 ? '1' : `\\frac{x^{${k}}}{${k}!}`}`);

                if (terms.length > 0 && terms[0].startsWith('+ ')) terms[0] = terms[0].substring(2);

            }

            if (terms.length === 0) return `P_{${n}}(x) = 0`;

            let originalFunc = func === 'sin' ? '\\sin(x)' : func === 'cos' ? '\\cos(x)' : 'e^x';

            return `${originalFunc} \\approx P_{${n}}(x) = ` + terms.join(' ');

        }

        // 4. 화면 업데이트 로직

        function initOrUpdatePlot(isFunctionChange = false) {

            const func = document.getElementById('funcSelect').value;

            const continuousN = parseFloat(document.getElementById('nSlider').value);

            const integerN = Math.floor(continuousN);

           

            document.getElementById('nValueDisplay').innerText = continuousN.toFixed(1);

            const funcNameMap = { 'sin': 'sin(x)', 'cos': 'cos(x)', 'exp': 'e^x' };

            if (!isPlotInitialized || isFunctionChange) {

                layout.yaxis.range = func === 'exp' ? [-2, 10] : [-3, 3];

                const yOriginal = getOriginalY(func, xValues);

                const traces = [

                    { ...traceOriginalTemplate, x: xValues, y: yOriginal, name: '원래 함수: ' + funcNameMap[func] },

                    { ...traceTaylorTemplate, x: xValues, y: ySmoothTaylor, name: `다항식 근사` }

                ];

                Plotly.react('plot', traces, layout, {responsive: true, displayModeBar: false});

                isPlotInitialized = true;

            } else {

                Plotly.restyle('plot', { 'y': [ySmoothTaylor] }, [1]);

            }

            if (lastRenderedN !== integerN || isFunctionChange) {

                const formulaStr = generatePolynomialString(func, integerN);

                const formulaElement = document.getElementById('formula');

                lastRenderedN = integerN;

            }

        }

            if(isPlotInitialized) Plotly.Plots.resize(document.getElementById('plot'));

        });

        document.getElementById('nSlider').addEventListener('input', () => initOrUpdatePlot(false));

        window.onload = () => setTimeout(() => initOrUpdatePlot(true), 100);
